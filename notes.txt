CREATE TYPE message_status AS ENUM (
    'PENDING',
    'FAILED',
    'RETRY'
);
CREATE TABLE rely (
    id BIGSERIAL PRIMARY KEY,
    topic VARCHAR(255) NOT NULL,
    payload BYTEA NOT NULL,
    codec VARCHAR(255) NOT NULL DEFAULT 'serializable',
    status message_status NOT NULL DEFAULT 'PENDING',
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    time_to_live INTEGER NOT NULL DEFAULT -1
);

---- Create notification function
CREATE OR REPLACE FUNCTION notify_rely_changes()
RETURNS trigger AS $$
BEGIN
    PERFORM pg_notify(
        'rely_changed',
        COALESCE(NEW.id)::text
    );

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

---- Apply over table rely
CREATE TRIGGER after_rely_changed
AFTER INSERT OR UPDATE
ON rely
FOR EACH ROW
EXECUTE FUNCTION notify_rely_changes();

---- List user defined function
select n.nspname as function_schema,
       p.proname as function_name,
       l.lanname as function_language,
       case when l.lanname = 'internal' then p.prosrc
            else pg_get_functiondef(p.oid)
            end as definition,
       pg_get_function_arguments(p.oid) as function_arguments,
       t.typname as return_type
from pg_proc p
left join pg_namespace n on p.pronamespace = n.oid
left join pg_language l on p.prolang = l.oid
left join pg_type t on t.oid = p.prorettype
where n.nspname not in ('pg_catalog', 'information_schema')
order by function_schema,
         function_name
